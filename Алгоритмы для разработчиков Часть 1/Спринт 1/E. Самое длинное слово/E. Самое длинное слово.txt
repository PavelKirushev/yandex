n = int(input())
s = input()
cnt = 0
cntmax = 0
slovo = ""
slovomax = ""
for i in range(n):
    if s[i] != " ":
        cnt += 1
        slovo += s[i]
    else:
        if cnt > cntmax:
            cntmax = cnt
            slovomax = slovo
        slovo = ""
        cnt = 0
print(slovomax)
print(cntmax)
